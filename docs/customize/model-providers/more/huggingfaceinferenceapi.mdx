## Configuration

<Tabs>
  <Tab title="YAML">
  ```yaml title="config.yaml"
  models:
    - name: <MODEL_NAME>
      provider: huggingface-inference-api
      model: <MODEL_ID>
      apiKey: <YOUR_HF_TOKEN>
      apiBase: <YOUR_HF_INFERENCE_API_ENDPOINT_URL>
  ```
  </Tab>
  <Tab title="JSON (Deprecated)">
  ```json title="config.json"
  {
    "models": [
      {
        "title": "<MODEL_NAME>",
        "provider": "huggingface-inference-api",
        "model": "<MODEL_ID>",
        "apiKey": "<YOUR_HF_TOKEN>",
        "apiBase": "<YOUR_HF_INFERENCE_API_ENDPOINT_URL>"
      }
    ]
  }
  ```
  </Tab>
</Tabs>
